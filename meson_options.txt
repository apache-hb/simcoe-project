# private package: shipping to testers
# release package: shipping to public

    description : 'Package type when building',
    choices : [ 'devel', 'private', 'release' ],
    value : 'devel'
)

    description : 'Trace compile time',
    value : 'disabled'
)

    description : 'Enable freetype logging',
    value : 'disabled'
)

option('enable_pix', type : 'feature',
    description : 'Enable PIX event runtime',
    value : 'auto'
)

option('warp_redist', type : 'feature',
    description : 'Enable searching for the new warp runtime redist',
    value : 'auto'
)
